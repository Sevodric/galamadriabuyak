start fight:
  shuffle decks
  WHILE player is alive AND monster is alive DO

    player's turn:
      draw until 3 cards in hand    
      WHILE cammand is not valid DO
        ask for an input
      END WHILE
      WHILE command is not "end turn" DO
        IF command is "help" DO
          display cards informations
        ELSE IF command is "use" DO
          apply card's effect(s)
        END IF
      END WHILE

    monster's turn:
      draw until 3 cards in hand
      perform monster's actions (AI)

  END WHILE
